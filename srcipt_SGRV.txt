CREATE TABLE `VehiculosReporte` (
  `idVehiculosReporte` int,
  `idReporte` int,
  `idVehiculo` int,
  PRIMARY KEY (`idVehiculosReporte`)
);

CREATE TABLE `AgenteDeTransito` (
  `idAgenteDeTransito` int,
  `nombre` varchar(255),
  PRIMARY KEY (`idAgenteDeTransito`)
);

CREATE TABLE `Conductor` (
  `idConductor` int,
  `nombre` varchar(255),
  `fechaNacimiento` Date,
  `numeroLicencia` varchar(255),
  `telefono` varchar(10),
  PRIMARY KEY (`idConductor`)
);

CREATE TABLE `Vehiculo` (
  `idVehiculo` int,
  `marca` varchar(255),
  `modelo` varchar(255),
  `año` varchar(4),
  `color` varchar(255),
  `nombreAseguradora` varchar(255),
  `polizaSeguro` varchar(255),
  `placas` varchar(255),
  `idConductor` int,
  PRIMARY KEY (`idVehiculo`)
);

CREATE TABLE `Fotografia` (
  `idFotografia` int,
  `rutaArchivo` varchar(255),
  `idReporte` int,
  PRIMARY KEY (`idFotografia`)
);

CREATE TABLE `Delegacion` (
  `idDelegacion` int,
  `nombre` varchar(255),
  `direccion` varchar(255),
  `codigoPostal` varchar(5),
  `municipio` varchar(255),
  `telefono` varchar(10),
  `correo` varchar(255),
  PRIMARY KEY (`idDelegacion`)
);

CREATE TABLE `Dictamen` (
  `Folio` int,
  `descripcion` varchar(250),
  `fechaYHora` Datetime,
  `idPerito` int,
  `idReporte` int,
  PRIMARY KEY (`Folio`)
);

CREATE TABLE `Soporte` (
  `idSoporte` int,
  `nombre` varchar(255),
  PRIMARY KEY (`idSoporte`)
);

CREATE TABLE `ConductoresReporte` (
  `idConductoresReporte` int,
  `idReporte` int,
  `idConductor` int,
  PRIMARY KEY (`idConductoresReporte`)
);

CREATE TABLE `Mensaje` (
  `idMensaje` int,
  `contenido` varchar(255),
  `fechaYHora` Datetime,
  `destinatario` int,
  `remitente` int,
  PRIMARY KEY (`idMensaje`)
);

CREATE TABLE `Reporte` (
  `idReporte` int,
  `fecha` Date,
  `direccion` varchar(255),
  `descripcion` varchar(255),
  `estado` varchar(255),
  `idAgenteDeTransiito` int,
  PRIMARY KEY (`idReporte`)
);

CREATE TABLE `Usuario` (
  `idUsaurio` int,
  `username` varchar(255),
  `contraseña` varchar(255),
  `cargo` varchar(255),
  `delegacion` int,
  PRIMARY KEY (`idUsaurio`)
);

CREATE TABLE `Administrativo` (
  `idAdministrativo` int,
  `nombre` varchar(255),
  PRIMARY KEY (`idAdministrativo`)
);

CREATE TABLE `Perito` (
  `idPerito` int,
  `nombre` varchar(255),
  PRIMARY KEY (`idPerito`)
);
